74static void controlLeds(void *pvParameter)

120static void controlTeclas(void *pvParameter)
121{
122 uint8_t teclas;
123 while (true)
124 {
125 teclas = SwitchesRead();
126 switch (teclas)
127 {
128 case SWITCH_1:
129 encendido = !encendido;
130 if (!encendido)
131 {
132 hold = false;
133 }
134 break;
135 case SWITCH_2:
136 hold = !hold;
137 default:
138 break;
139 }
140 vTaskDelay(CONFIG_BLINK_PERIOD_TECLAS / portTICK_PERIOD_MS);
141 }
