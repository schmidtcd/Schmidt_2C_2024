162static void lecturaAcelerometro(void *pvParameter)
163{
164 uint16_t acelerometro_1 = 0;
165 uint16_t acelerometro_2 = 0;
166 uint16_t acelerometro_3 = 0;
167 float aceleracion_x;
168 float aceleracion_y;
169 float aceleracion_z;
170 float suma;
171 while (true)
172 {
173 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
174 AnalogInputReadSingle(CH1, &acelerometro_1);
175 AnalogInputReadSingle(CH2, &acelerometro_2);
176 AnalogInputReadSingle(CH3, &acelerometro_3);
177 aceleracion_x = ((float)acelerometro_1 - 1650) / 0.3;
178 aceleracion_y = ((float)acelerometro_2 - 1650) / 0.3;
179 aceleracion_z = ((float)acelerometro_3 - 1650) / 0.3;
180 suma = (aceleracion_x + aceleracion_y + aceleracion_z) / 1000;
181 if (suma > 4)
182 {
183 UartSendString(UART_CONNECTOR, "Caída detectada\r\n");
184 }
185 }
