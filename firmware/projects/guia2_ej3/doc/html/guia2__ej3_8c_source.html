188void uartControl(void *param)
189{
190 uint8_t tecla;
191 UartReadByte(UART_PC, &tecla);
192 if (tecla == 'O' || tecla == 'o')
193 {
194 encendido = !encendido;
195 if (!encendido)
196 {
197 hold = false;
198 }
199 }
200 else if (tecla == 'H' || tecla == 'h')
201 {
202 if (encendido)
203 {
204 hold = !hold;
